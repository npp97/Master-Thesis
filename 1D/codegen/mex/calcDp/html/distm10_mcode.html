1function D = distm(X,Y)
 2
 3% DISTM   Distance matrix.
 4%  DISTM(X,Y) is the matrix of pairwise euclidean
 5%  distances between the points in the rows of X and Y
 6%  code by Paritosh Mokhasi and Gregory Fasshauer
 7
 8[M,s] = size(X); [N,s] = size(Y);
 9D = repmat(sum(X.*X,2),1,N) - 2*X*Y' + repmat((sum(Y.*Y,2))',M,1);
10D = real(sqrt(complex(D)));
11
12end

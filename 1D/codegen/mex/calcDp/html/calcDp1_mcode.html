1function Dp = calcDp(X,Xp,rstar,Dzero)
 2    assert( isa(X,'double'))
 3    assert( isa(Xp,'double'))
 4    assert( isa(rstar,'double'))
 5    assert( isa(Dzero,'double'))
 6    Dp = Dzero*ones(size(X,1),1);
 7    for j=1:size(X,1)
 8        Dp_old = -inf;
 9        k=0;
10        while((((abs(Dp_old-Dp(j)))/Dp(j))>1e-2) && k < 15)
11            Dp_old = Dp(j);
12            ind = (distm(X(j,:),Xp)<=rstar*Dp(j,:));
13            if(sum(ind)>0)
14                Dp(j) = min(monitor_function(Xp(ind,:),Dzero));
15                k=k+1;
16            else
17                Dp(j) = monitor_function(X(j,:));
18                break;
19            end
20        end
21    end
22end
